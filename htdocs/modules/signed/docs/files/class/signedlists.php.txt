    class signedLists extends XoopsLists

	static function getMonthsList() {
		signed_loadLanguage('months');
		return array(
				'0' => '--------',
				'1' => _MONTH_ONE ,
				'2' => _MONTH_TWO ,
				'3' => _MONTH_THREE ,
				'4' => _MONTH_FOUR ,
				'5' => _MONTH_FIVE ,
				'6' => _MONTH_SIX ,
				'7' => _MONTH_SEVEN ,
				'8' => _MONTH_EIGHT ,
				'9' => _MONTH_NINE ,
				'10' => _MONTH_TEN ,
				'11' => _MONTH_ELEVEN ,
				'12' => _MONTH_TWELVE);
	}
		
        static function getYearsList()
        {
        	signed_loadLanguage('global');
        
        	$years_list = array(0 => '----');
        	for($i=date('Y');$i<=date('Y') + constant('_YEARS_NUMBEROF'); $i++)
        		$years_list[$i] = $i;
        
        	return $years_list;
        }
    }
